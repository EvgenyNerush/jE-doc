753 bks_synchrotron_td( double ri
754  , double m
755  , double b
756  , double gamma_p
757  ) {
758  // The normalization can be obtained analogous to jackson1483_num, see comments there. Note
759  // that the generated photons are evenly distributed around the electron trajectory (which is a
760  // circle), i.e. they are evenly distributed in the angle $ \phi $ which in turn corresponds to
761  // the direction of the photon wavevector in the plane of the electron trajectory. Thus, the
762  // solid angle is $ d\Omega = \cos \theta \, d\phi d\theta $, and for the emission probability
763  // we get $ d^2 W / d\theta d\omega = 2 W_m \cos \theta \omega^2 / \pi^2 $ with $ W_m $ the
764  // emission probability computed with bks_synchrotron_emission_probability function.

766  [=]( std::tuple<double, double> theta_omega ) {

768  double omega = std::get<1>(theta_omega);
769  if (omega > 0 and omega * b < m * gamma_p) {
770  return 2 * pow(omega / M_PI, 2) * cos(theta) / m
